import java.util.Scanner;

public class code3{
    public static void main(String[]args){
    Scanner sc=new Scanner(System.in);

    System.out.println("Enter SellingPrice");
    float SellingPrice=sc.nextFloat();

    System.out.println("Enter DiscountedPercentage");
    float DiscountedPercentage=sc.nextFloat();

    Float calculation=SellingPrice-(DiscountedPercentage/100*SellingPrice);
        
     System.out.println("Calculation="+calculation);
   
     sc.close();
    }

    }

}
